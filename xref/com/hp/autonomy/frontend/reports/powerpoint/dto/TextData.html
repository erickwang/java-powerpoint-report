8   import lombok.AllArgsConstructor;
9   import lombok.Data;
10  import lombok.NoArgsConstructor;
11  
12  /**
13   * DTO to represent text data to use in reports.
14   */
15  @Data
16  @NoArgsConstructor
17  @AllArgsConstructor
18  public class TextData implements ComposableElement {
19  
20      /**
21       * Array of individual text elements. There isn't an implicit newline after each block of text, to allow you to
22       * put differently-formatted text elements adjacent to each other.
23       */
24      private Paragraph[] text;
25  
26      /**
27       * DTO representing a single block of text with specific formatting.
28       */
29      @Data
30      @NoArgsConstructor
31      public static class Paragraph {
32  
33          /** Whether to render the text in bold. */
34          private boolean bold;
35  
36          /** Whether to render the text in italic. */
37          private boolean italic;
38  
39          /** The text to write. You can insert '\n' newlines for newlines. There isn't an implicit newline at the end. */
40          private String text = "\n";
41  
42          /** The font colour. */
43          private String color = "#000000";
44  
45          /** The font size. */
46          private double fontSize = 12;
47  
48          public Paragraph(final boolean bold, final boolean italic, final String text, final String color, final double fontSize) {
49              this.bold = bold;
50              this.italic = italic;
51              this.text = text;
52              this.color = color;
53              this.fontSize = fontSize;
54          }
55      }

57  }
