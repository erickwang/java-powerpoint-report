6   package com.hp.autonomy.frontend.reports.powerpoint.dto;
7   
8   import lombok.AllArgsConstructor;
9   import lombok.Data;
10  import lombok.NoArgsConstructor;
11  
12  /**
13   * DTO to represent a table of data.
14   */
15  @Data
16  @NoArgsConstructor
17  @AllArgsConstructor
18  public class TableData implements ComposableElement {
19  
20      /** Number of rows. */
21      private int rows;
22  
23      /** Number of columns. */
24      private int cols;
25  
26      /** Contents of the table in order, there should be {@code rows * cols} cells. List values from the first row, then the second row, and so on. */
27      private String[] cells;
28  
29      public boolean validateInput() {
30          return rows > 0 && cols > 0 && cells != null && cells.length == rows * cols;
31      }
32  }
