35      private static final String RELATION_NAMESPACE = "http://schemas.openxmlformats.org/officeDocument/2006/relationships";

37      /** Parsed PowerPoint file from the template. */
38      private final XMLSlideShow pptx;
39      /** Doughnut chart XML object, cached so we can clone it. */
40      private final ImmutablePair<XSLFChart, CTGraphicalObjectFrame> doughnutChart;
41      /** An xy scatterplot chart XML object, cached so we can clone it.  */
42      private final ImmutablePair<XSLFChart, CTGraphicalObjectFrame> graphChart;
43  
44      SlideShowTemplate(final InputStream inputStream) throws TemplateLoadException {
45          try {
46              // There should be a doughnut chart in slide 1 and a scatterplot chart in slide 2
47              pptx = new XMLSlideShow(inputStream);
48  
49              final List<XSLFSlide> slides = pptx.getSlides();
50  
51              if (slides.size() != 2) {
52                  throw new TemplateLoadException("Template powerpoint should have two slides, doughnut chart on slide 1 and time-axis xy scatterplot chart on slide 2");
53              }
54  
55              XSLFSlide slide = slides.get(0);
56  
57              doughnutChart = getChart(slide, "First slide should have a doughnut chart");
58  
59              if (ArrayUtils.isEmpty(doughnutChart.getLeft().getCTChart().getPlotArea().getDoughnutChartArray())) {
60                  throw new TemplateLoadException("First slide has the wrong chart type, should have a doughnut chart");
61              }
62  
63              graphChart = getChart(slides.get(1), "Second slide should have a time-axis xy scatterplot chart");
64  
65              if (ArrayUtils.isEmpty(graphChart.getLeft().getCTChart().getPlotArea().getScatterChartArray())) {
66                  throw new TemplateLoadException("Second slide has the wrong chart type, should have a time-axis xy scatterplot chart");
67              }
68  
69              // Remove the slides afterwards
70              pptx.removeSlide(1);
71              pptx.removeSlide(0);
72          }
73          catch(IOException e) {
74              throw new TemplateLoadException("Error while loading slide show", e);
75          }
76      }

126     /**
127      * Given an existing slide, search its relations to find a chart object.
128      * @param slide a slide from the template.
129      * @param error if we can't find a slide, this error message will be returned as the exception.
130      * @return a pair containing the chart.xml data and the graphical object which represented it on the slide.
131      * @throws TemplateLoadException if we can't find a chart object.
132      */
133     private ImmutablePair<XSLFChart, CTGraphicalObjectFrame> getChart(final XSLFSlide slide, final String error) throws TemplateLoadException {
134         for(POIXMLDocumentPart.RelationPart part : slide.getRelationParts()) {
135             if (part.getDocumentPart() instanceof XSLFChart) {
136                 final String relId = part.getRelationship().getId();
137 
138                 for(XSLFShape shape : slide.getShapes()) {
139                     if (shape instanceof XSLFGraphicFrame) {
140                         final CTGraphicalObjectFrame frameXML = (CTGraphicalObjectFrame) shape.getXmlObject();
141                         final XmlObject[] children = frameXML.getGraphic().getGraphicData().selectChildren(new QName(XSSFRelation.NS_CHART, "chart"));
142 
143                         for(final XmlObject child : children) {
144                             final String imageRel = child.getDomNode().getAttributes().getNamedItemNS(RELATION_NAMESPACE, "id").getNodeValue();
145 
146                             if (relId.equals(imageRel)) {
147                                 return new ImmutablePair<>(part.getDocumentPart(), frameXML);
148                             }
149                         }
150                     }
151                 }
152             }
153         }
154 
155         throw new TemplateLoadException(error);
156     }

158     /**
159      * Utility function to clone the graphical object which represents a chart on a slide.
160      * @param base the object to clone.
161      * @param relId the new relation ID we should insert.
162      * @param shapeId the new shape ID we should insert.
163      * @param shapeName the new shape name we should insert.
164      * @param anchor the bounds of the new shape object in PowerPoint coordinates, if set, or null to use the existing clone's bounds.
165      * @return a new clone object with the desired properties.
166      */
167     private CTGraphicalObjectFrame cloneShapeXML(final CTGraphicalObjectFrame base, final String relId, final int shapeId, final String shapeName, final Rectangle2D.Double anchor) {
168         /* Based on viewing the raw chart.
169           <p:graphicFrame>
170             <p:nvGraphicFramePr>
171               <p:cNvPr id="4" name="Chart 3"/>
172               <p:cNvGraphicFramePr/>
173               <p:nvPr>
174                 <p:extLst>
175                   <p:ext uri="{D42A27DB-BD31-4B8C-83A1-F6EECF244321}">
176                     <p14:modId xmlns:p14="http://schemas.microsoft.com/office/powerpoint/2010/main" val="866141002"/>
177                   </p:ext>
178                 </p:extLst>
179               </p:nvPr>
180             </p:nvGraphicFramePr>
181             <p:xfrm>
182               <a:off x="0" y="0"/>
183               <a:ext cx="12192000" cy="6858000"/>
184             </p:xfrm>
185             <a:graphic>
186               <a:graphicData uri="http://schemas.openxmlformats.org/drawingml/2006/chart">
187                 <c:chart xmlns:c="http://schemas.openxmlformats.org/drawingml/2006/chart"
188                          xmlns:r="http://schemas.openxmlformats.org/officeDocument/2006/relationships" r:id="rId2"/>
189               </a:graphicData>
190             </a:graphic>
191           </p:graphicFrame>
192          */
193         final CTGraphicalObjectFrame copy = (CTGraphicalObjectFrame) base.copy();
194 
195         final CTNonVisualDrawingProps cNvPr = copy.getNvGraphicFramePr().getCNvPr();
196         cNvPr.setName(shapeName);
197         cNvPr.setId(shapeId);
198 
199         final XmlObject[] children = copy.getGraphic().getGraphicData().selectChildren(new QName(XSSFRelation.NS_CHART, "chart"));
200 
201         if (anchor != null) {
202             final CTPoint2D off = copy.getXfrm().getOff();
203             off.setX((int) (anchor.getX() * EMU_PER_POINT));
204             off.setY((int) (anchor.getY() * EMU_PER_POINT));
205 
206             final CTPositiveSize2D ext = copy.getXfrm().getExt();
207             ext.setCx((int) (anchor.getWidth()* EMU_PER_POINT));
208             ext.setCy((int) (anchor.getHeight() * EMU_PER_POINT));
209         }
210 
211         for(final XmlObject child : children) {
212             child.getDomNode().getAttributes().getNamedItemNS(RELATION_NAMESPACE, "id").setNodeValue(relId);
213         }
214 
215         return copy;
216     }
