5   package com.hp.autonomy.frontend.reports.powerpoint;
6   
7   import org.apache.commons.codec.binary.Base64;
8   import org.apache.poi.sl.usermodel.PictureData;
9   
10  /**
11   * Default implementation of ImageSource; treats all image identifiers as base64-encoded data URLs.
12   * If the identifier is missing a data: prefix, it'll be assumed to be JPEG.
13   */
14  public class DataUriImageSource implements ImageSource {
15  
16      private static final String DATA_PREFIX = "data:";
17  
18      /**
19       * Accepts base64-encoded image data and converts them to image data.
20       * @param imageId the image identifier.
21       * @return image data corresponding to the image.
22       * @throws IllegalArgumentException if we can't fetch the image.
23       */
24      @Override
25      public ImageData getImageData(final String imageId) throws IllegalArgumentException {
26          // assume it's base64 data
27          final String contentType, data;
28  
29          if(imageId.startsWith(DATA_PREFIX)) {
30              // e.g. 'data:image/png;base64,iVBOR....'
31              contentType = imageId.substring(DATA_PREFIX.length(), imageId.indexOf(';'));
32              data = imageId.substring(imageId.indexOf(","));
33          }
34          else {
35              // assume it's the data part of the base64-encoded JPEG, since Find sends list thumbnails this way
36              contentType = "image/jpeg";
37              data = imageId;
38          }
39  
40          for(final PictureData.PictureType pictureType : PictureData.PictureType.values()) {
41              if(pictureType.contentType.equalsIgnoreCase(contentType)) {
42                  return new ImageData(pictureType, Base64.decodeBase64(data));
43              }
44          }
45  
46          throw new IllegalArgumentException("Unable to parse data URI: " + imageId);
47      }
48  }
