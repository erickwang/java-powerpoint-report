6   package com.hp.autonomy.frontend.reports.powerpoint;
7   
8   import com.hp.autonomy.frontend.reports.powerpoint.dto.DategraphData;
9   import com.hp.autonomy.frontend.reports.powerpoint.dto.ListData;
10  import com.hp.autonomy.frontend.reports.powerpoint.dto.MapData;
11  import com.hp.autonomy.frontend.reports.powerpoint.dto.ReportData;
12  import com.hp.autonomy.frontend.reports.powerpoint.dto.SunburstData;
13  import com.hp.autonomy.frontend.reports.powerpoint.dto.TableData;
14  import com.hp.autonomy.frontend.reports.powerpoint.dto.TopicMapData;
15  import org.apache.poi.xslf.usermodel.XMLSlideShow;
16  
17  /**
18   * Primary interface to the PowerPoint API.
19   * @see <a href="https://github.com/hpe-idol/java-powerpoint-report/" target="_blank">README.md</a> for examples and usage instructions.
20   */
21  public interface PowerPointService {
22  
23      /**
24       * Validates the template used to see if it has the correct slides.
25       * @throws TemplateLoadException if the template is invalid.
26       */
27      void validateTemplate() throws TemplateLoadException;
28  
29      /**
30       * Renders a topic map as a PowerPoint presentation.
31       * @param topicmap the data representing the topic map paths.
32       * @return a PowerPoint presentation with a single slide.
33       * @throws TemplateLoadException if the template is invalid.
34       */
35      XMLSlideShow topicmap(TopicMapData topicmap) throws TemplateLoadException;
36  
37      /**
38       * Renders a sunburst as a PowerPoint presentation. Only one level of sunburst is supported for now,
39       * since it's actually represented by a doughnut chart (based on the first slide in the template).
40       * @param sunburst the data representing the sunburst.
41       * @return a PowerPoint presentation with a single slide.
42       * @throws TemplateLoadException if the template is invalid.
43       */
44      XMLSlideShow sunburst(SunburstData sunburst) throws TemplateLoadException;
45  
46      /**
47       * Renders a table as a PowerPoint presentation.
48       * @param tableData the data representing the table.
49       * @param title an optional title.
50       * @return a PowerPoint presentation with a single slide.
51       * @throws TemplateLoadException if the template is invalid.
52       */
53      XMLSlideShow table(TableData tableData, String title) throws TemplateLoadException;
54  
55      /**
56       * Renders a map as a PowerPoint presentation.
57       * @param map the data representing the map.
58       * @param title an optional title.
59       * @return a PowerPoint presentation with a single slide.
60       * @throws TemplateLoadException if the template is invalid.
61       */
62      XMLSlideShow map(MapData map, String title) throws TemplateLoadException;
63  
64      /**
65       * Renders a list of documents as a PowerPoint presentation, similar to how they're rendered in Find.
66       * We only support thumbnail images if they're provided as base64 strings to embed into the document,
67       *   since the POI API doesn't support links to external images.
68       * @param documentList the data containing all the documents
69       * @param results an optional string which will be shown in the top-left.
70       * @param sortBy an optional string which will be shown in the top-right.
71       * @return a PowerPoint presentation with the list results paginated over multiple slides.
72       * @throws TemplateLoadException if the template is invalid.
73       */
74      XMLSlideShow list(ListData documentList, String results, String sortBy) throws TemplateLoadException;
75  
76      /**
77       * Renders a date graph as a PowerPoint presentation.
78       * This is drawn based on the xy scatterplot chart (which must be contained in the second slide in the template).
79       * @param data the data representing the graph series to plot.
80       * @return a PowerPoint presentation with a single slide.
81       * @throws TemplateLoadException if the template is invalid.
82       */
83      XMLSlideShow graph(DategraphData data) throws TemplateLoadException;
84  
85      /**
86       * Render a composite report consisting of multiple visualizations as a PowerPoint presentation.
87       * @param report the data containing a list of visualizations to render.
88       * @param slidePerVisualizer if true, we'll put each visualization on a separate slide.
89       * @return a PowerPoint presentation with a single slide.
90       * @throws TemplateLoadException if the template is invalid.
91       */
92      XMLSlideShow report(ReportData report, final boolean slidePerVisualizer) throws TemplateLoadException;
93  
94  }
