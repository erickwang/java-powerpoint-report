7   /**
8    * An interface to convert image identifiers into actual image data for embedding into PowerPoint.
9    *
10   * The default implementation (DataUriImageSource) only handles data URIs, but if you have a more compact representation
11   *   to uniquely identify your images, you could provide your own custom ImageSource class to convert your compact image
12   *   identifiers into actual image data; which would save some round-trip data.
13   *
14   * If you have a mix of HTTP/HTTPs links and data uris, you can use WebAndDataUriImageSource. You may want to override
15   *   WebAndDataUriImageSource.allowHttpURI which only allows URL paths ending in '.jpeg', '.jpg', '.png' or '.gif'
16   *   with your own whitelist.
17   *
18   *  @see DataUriImageSource
19   *  @see WebAndDataUriImageSource
20   */
21  public interface ImageSource {
22  
23      /**
24       * Converts an image identifier into image data.
25       * @param imageId the image identifier.
26       * @return image data corresponding to the image.
27       * @throws IllegalArgumentException if we can't fetch the image.
28       */
29      ImageData getImageData(final String imageId) throws IllegalArgumentException;
30  
31      ImageSource DEFAULT = new DataUriImageSource();
32  
33  }
